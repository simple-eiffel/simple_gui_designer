class interface
	SIMPLE_WEB_CLAUDE_CLIENT

create 
	make

feature -- Access

	api_key: STRING_8
			-- Anthropic API key

	model: STRING_8
			-- Claude model to use

	max_tokens: INTEGER_32
			-- Maximum tokens in response
	
feature -- Settings

	set_model (a_model: STRING_8)
			-- Set Claude model
		require
			model_attached: a_model /= Void
			model_not_empty: not a_model.is_empty
		ensure
			model_set: model = a_model

	set_max_tokens (a_max: INTEGER_32)
			-- Set maximum tokens
		require
			max_positive: a_max > 0
		ensure
			max_set: max_tokens = a_max
	
feature -- Operations

	send_message (a_user_message: STRING_8): detachable STRING_8
			-- Send message to Claude and return response
		require
			message_attached: a_user_message /= Void
			message_not_empty: not a_user_message.is_empty

	send_message_with_system (a_user_message, a_system_prompt: STRING_8): detachable STRING_8
			-- Send message with system prompt
		require
			message_attached: a_user_message /= Void
			message_not_empty: not a_user_message.is_empty
			system_attached: a_system_prompt /= Void
	
invariant
	api_key_attached: api_key /= Void
	client_attached: client /= Void
	model_attached: model /= Void
	max_tokens_positive: max_tokens > 0
